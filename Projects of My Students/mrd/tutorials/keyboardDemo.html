//keyboardDemo
//demonstrates two main flavors of keyboard input
// Andy Harris, 4/05

_root.onEnterFrame = function(){
  
  output = "";
  if(Key.isDown(Key.LEFT)){
    output = "left";
  } // end if

  if(Key.isDown(Key.RIGHT)){
    output = "right";
  } // end if

  if(Key.isDown(Key.UP)){
    output = "up";
  } // end if

  if(Key.isDown(Key.DOWN)){
    output = "down";
  } // end if

  if(Key.isDown(Key.SPACE)){
    output = "space";
  } // end if

  if(Key.isDown(65)){
    output = "A";
  } // end if

  lastPressed = String.fromCharCode(Key.getAscii());
  
} // end if
