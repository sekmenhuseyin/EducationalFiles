//check operation
switch(op){
  case "+":
    X = Math.ceil(Math.random() * 10);
    Y = Math.ceil(Math.random() * 10);
    correct = X + Y;
    trace(correct);
  break;
  case "-":
    Y = Math.ceil(Math.random() * 10);
    correct = Math.ceil(Math.random() * 10);
    X = correct + Y;
    trace(correct);
  break;
  case "*":
    X = Math.ceil(Math.random() * 10);
    Y = Math.ceil(Math.random() * 10);
    correct = X * Y;
    trace(correct);
  break;
  case "/":
    Y = Math.ceil(Math.random() * 10);
    correct = Math.ceil(Math.random() * 10);
    X = correct * Y;
    trace(correct);
  break;
  default:
    trace("There's a problem here");
  break;
} // end switch
