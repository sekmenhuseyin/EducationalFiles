// binaryDice
// demonstrates switch structure
// Andy Harris, 12/04

btnRoll.onRelease = function(){
  //roll a die
  die = Math.ceil(Math.random() * 6);

  switch(die){
    case 1:
      output = "001";
      break;
    case 2:
      output = "010";
      break;
    case 3:
      output = "011";
      break;
    case 4:
      output = "100";
      break;
    case 5:
      output = "101";
      break;
    case 6:
      output = "110";
      break;
    default:
      output = "problem...";
      break;
  } // end switch
} // end roll
