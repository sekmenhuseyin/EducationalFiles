// sixOrNot
// demonstrates if-else
// Andy Harris, 12/04

btnRoll.onRelease = function(){
  //roll a die
  die = Math.ceil(Math.random() * 6);

  //check to see if it's a 6
  if (die == 6){
    output = "You got a six!";
  } else {
    output = "That's not a six.";
  } // end if
} // end roll
